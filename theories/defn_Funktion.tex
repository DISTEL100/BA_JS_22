\newglossaryentry{g:Funktion}{
	name=Funktion,
	description={
		Ist eine Operation, die eine feste Anzahl Konstanten einer Ausgangsmenge in eine Konstante einer Zielmenge überführt. Die Anzahl ist die Stelligkeit der Funktion.
	}
}

\begin{defn}[\gls{g:Funktion}]
	Ist eine Ausgangsmenge von Konstanten, eine Zielmenge von Konstanten, eine Stelligkeit $n$ und eine Zuordnung von jeweils $n$-Elemnten der Ausgangsmenge zu einem Element der Zielmenge.\cite[15]{ACIntro}
\end{defn}
